
=== append a new text after the cursor ===
-----------------------------------------------------------------------
"a"
-----------------------------------------------------------------------

=== insert a new text before the cursor ===
-----------------------------------------------------------------------
"i"
-----------------------------------------------------------------------

=== new line below the current string ===
-----------------------------------------------------------------------
"o"
-----------------------------------------------------------------------

=== new line above the current string ===
-----------------------------------------------------------------------
"O" - (SHIFT + o)
-----------------------------------------------------------------------

=== one word forward ===
-----------------------------------------------------------------------
"w"
-----------------------------------------------------------------------

=== one word backward ===
-----------------------------------------------------------------------
"b"
-----------------------------------------------------------------------

=== to the end of the current word ===
-----------------------------------------------------------------------
"e"
-----------------------------------------------------------------------

=== at the beginning of the current string ===
-----------------------------------------------------------------------
"0"
-----------------------------------------------------------------------

=== to the end of the string ===
-----------------------------------------------------------------------
"$"
-----------------------------------------------------------------------

=== to the end of the current document ===
-----------------------------------------------------------------------
"G"
-----------------------------------------------------------------------

=== at the beginning of the string  ===
(without spaces)
-----------------------------------------------------------------------
"^"
-----------------------------------------------------------------------

=== at the beginning of the screen ===
-----------------------------------------------------------------------
"H" - (SHIFT + h)
-----------------------------------------------------------------------

=== to the end of the screen ===
-----------------------------------------------------------------------
"L" - (SHIFT + l)
-----------------------------------------------------------------------

===========================
5j - to move 5 lines down;
3h - to jump 3 characters left;
===========================

=== to delete text from the cursor to the end
of the string ===
-----------------------------------------------------------------------
"D"
-----------------------------------------------------------------------

=== to join the next lines ===
-----------------------------------------------------------------------
"J"
-----------------------------------------------------------------------

=== to delete the entire line ===
-----------------------------------------------------------------------
"dd"
-----------------------------------------------------------------------

=== to delete the current word ===
-----------------------------------------------------------------------
"dw"
-----------------------------------------------------------------------

=== to replace the character ===
-----------------------------------------------------------------------
"R"
-----------------------------------------------------------------------

=== to replace the string ===
(begins an empty line at the beginning)
-----------------------------------------------------------------------
"S"
-----------------------------------------------------------------------

=== to delete the current character ===
(the cursor is left on the same position)
-----------------------------------------------------------------------
"s"
-----------------------------------------------------------------------

=== to undo recent changes ===
-----------------------------------------------------------------------
"u"
-----------------------------------------------------------------------

=== to change the register of the character ===
-----------------------------------------------------------------------
"~"
-----------------------------------------------------------------------

=== to search the word where the current cursor is ===
-----------------------------------------------------------------------
"*"
-----------------------------------------------------------------------

=== to change the color scheme ===
-----------------------------------------------------------------------
colorscheme default|evening|blue|desert
-----------------------------------------------------------------------

=== to switch on/off syntax highlighting ===
-----------------------------------------------------------------------
syntax off/on
-----------------------------------------------------------------------

=== to save configs for VIM ===
-----------------------------------------------------------------------
~/.vimrc
-----------------------------------------------------------------------

=== to highlight search results in VIM ===
-----------------------------------------------------------------------
set hlsearch
-----------------------------------------------------------------------

=== to disable highlight search in VIM ===
-----------------------------------------------------------------------
set hlsearch&
-----------------------------------------------------------------------

=== for increamental search ===
-----------------------------------------------------------------------
set incsearch
-----------------------------------------------------------------------

=== woking with tabs ===
-----------------------------------------------------------------------
tabs //list of tabs
tabn //next tab
tabp //previous tab
tab sp (split) //to split current tab
Vex //file's tree
<C-w-hjkl> //navigation through tabs
-----------------------------------------------------------------------

=== list of available options (autocompletion) ===
-----------------------------------------------------------------------
colorscheme Ctrl+d // to list all available schemes
help Ctrl +d // to show all help options
-----------------------------------------------------------------------

=== to set 'tab space' containing 4 spaces ===
-----------------------------------------------------------------------
set tabstop=4
-----------------------------------------------------------------------

=== to set color for line numbers ===
-----------------------------------------------------------------------
:highlight LineNr ctermfg=grey // for vim
:highlight LineNr guifg=#050505 // for gvim
// location for color schemes:
// /usr/share/vim/vim72/colors
-----------------------------------------------------------------------
